A jsx-free, super-lightweight and zero-dependency library to add interactivity to the web — without all the nonsense.

import {mount, tags} from 'tent';

const {div, button} = tags;

const Counter = {
  state: {count: 0},
  view: ({state}) => div([
    button('inc', {
      onclick: () => state.count++
    }),
    div(state.count),
    button('dec', {
      onclick: () => state.count--
    }),
  ])
}

mount(document.body, Counter);
